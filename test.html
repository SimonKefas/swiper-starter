// Randomize tilt + start autoplay
document.addEventListener('DOMContentLoaded', () => {
  setTimeout(() => {
    const el = document.querySelector('#marquee-gallery');
    const swiper = el?._swiperInstance;
    if (!el || !swiper) return;

    el.querySelectorAll('.swiper-slide').forEach((slide) => {
      const tilt = (Math.random() * 10 - 5).toFixed(2);
      const lift = (Math.random() * 14 - 7).toFixed(0);
      slide.style.setProperty('--tilt', `${tilt}deg`);
      slide.style.setProperty('--lift', `${lift}px`);
    });

    swiper.autoplay?.start();
  }, 100);
});

// Hook into the Swiper instance after init
document.addEventListener('DOMContentLoaded', () => {
  // Wait for Swiper Starter to initialize
  setTimeout(() => {
    const container = document.querySelector('#scale-slider');
    const swiper = container._swiperInstance;
    
    if (!swiper) return;
    
    // Function to update position classes
    function updatePositionClasses() {
      const slides = container.querySelectorAll('.swiper-slide');
      const activeIndex = swiper.activeIndex;
      
      slides.forEach((slide, i) => {
        // Remove all position classes
        slide.classList.remove(
          'pos-active', 'pos-prev-1', 'pos-prev-2',
          'pos-next-1', 'pos-next-2'
        );
        
        // Add class based on distance from active
        const diff = i - activeIndex;
        if (diff === 0) slide.classList.add('pos-active');
        else if (diff === -1) slide.classList.add('pos-prev-1');
        else if (diff === 1) slide.classList.add('pos-next-1');
        else if (diff === -2) slide.classList.add('pos-prev-2');
        else if (diff === 2) slide.classList.add('pos-next-2');
      });
    }
    
    // Run on init and slide change
    updatePositionClasses();
    swiper.on('slideChange', updatePositionClasses);
  }, 100);
});

All Data Attributes

Complete reference of available configuration options.

data-loop-mode="true"
Enable infinite loop mode
data-autoplay="3000"
Autoplay delay in ms, or "false" to disable
data-slider-duration="500"
Transition speed in ms
data-effect="fade"
Transition effect (slide, fade)
data-crossfade="true"
Enable crossfade for fade effect
data-slides-per-view="3"
Slides visible, or "auto"
data-space-between="16"
Gap between slides in px
data-single-slide="true"
Force 1 slide, disable breakpoints
data-breakpoints="{...}"
Custom responsive breakpoints (JSON)
data-progress-bar="true"
Top-level progress bar
data-bullet-progress="true"
Progress fill within bullets
data-custom-slider="true"
Enable range input navigation
data-slider-color="#hex"
Custom color for progress/slider
data-centered-slides="true"
Center active slide
data-free-mode="true"
Free scrolling without snap
data-slides-per-group="2"
Slides to advance per nav click
data-autoplay-inview="true"
Autoplay only when visible
data-disable-navigation="true"
Hide prev/next arrows
data-disable-pagination="true"
Hide pagination bullets
data-disable-touch="true"
Disable drag/swipe (wheel works)
data-full-height="true"
Slides fill container height
data-observer="true"
Watch for DOM mutations
data-disable-below="768"
Disable slider below viewport width
data-disable-above="991"
Disable slider above viewport width
